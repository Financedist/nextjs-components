### Clearable

Automatically show a clear button if a value is entered.

<Spacer />

<Editor
  scope={{ Container, ClearableInput,Search }}
  code={`() => {
const [value, setValue] = React.useState('initial')
return (<Container left>
  <Container>
    <ClearableInput value={value} onChange={e => setValue(e.target.value)} />
  </Container>

  <Container>
    <ClearableInput
      prefix={<Search />}
    />
  </Container>
</Container>)
}`}
/>
